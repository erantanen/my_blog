echo "enter directory, press Enter"
read dir
array=($(ls  "$dir"))
echo -e "-----------------------\n"

for elm in "${array[@]}";
  do
    cat_elm=$(echo ${dir}/${elm})
    if [ -d "${cat_elm}" ]
    then
      output= eval du -sh ${cat_elm}
      echo -n "$output"|tr '\n' ' '
     fi
   done

echo "enter word to find, press Enter"
read word
array=($(locate  "$word"))
echo -e "-----------------------\n"

for elm in "${array[@]}";
  do
   output= eval  ls -lh "$elm"
   echo "$output" |tr '\n' ' '
  done
